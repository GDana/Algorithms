## 내가 쓴 풀이


<details>
<summary>코드</summary>
<div markdown="1">     

```
class Solution {
    public long solution(int a, int b) {
        long answer = 0;
        int big = 0;
        int small = 0;

        if(a > b) {
            big = a;
            small = b;
        }else if(a == b) {
            return answer = a;
        }else {
            big = b;
            small = a;
        }

        for(int i = small; i < big+1; i++) {
            answer += i;
        }

        return answer;
    }
}
```

```
class Solution {

    public long solution(int a, int b) {
        return sumAtoB(Math.min(a, b), Math.max(b, a));
    }

    private long sumAtoB(long a, long b) {
        return (b - a + 1) * (a + b) / 2;
    }
}
```
